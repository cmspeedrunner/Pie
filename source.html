  219.         def setup(self):

  221.             self.old_y = None
  222.             self.line_width = self.choose_size_button.get()

  227.             self.c.bind('<ButtonRelease-1>', self.reset)
  228.  
  229.         def use_pen(self):

  235.         def choose_color(self):
  236.             self.eraser_on = False
  237.             self.color = tkinter.colorchooser.askcolor(color=self.color)[1]
  238.  
  239.         def use_eraser(self):

  242.         def activate_button(self, some_button, eraser_mode=False):
  243.             self.active_button.config(relief=RAISED)
  244.             some_button.config(relief=SUNKEN)

  247.  
  248.         def paint(self, event):
  249.             self.line_width = self.choose_size_button.get()
  250.             paint_color = 'white' if self.eraser_on else self.color
  251.             if self.old_x and self.old_y:
  252.                 self.c.create_line(self.old_x, self.old_y, event.x, event.y,
  253.                                    width=self.line_width, fill=paint_color,
  254.                                    capstyle=ROUND, smooth=TRUE, splinesteps=36)
  255.             self.old_x = event.x
  256.             self.old_y = event.y
  257.  
  258.         def reset(self, event):
  259.             self.old_x, self.old_y = None, None
